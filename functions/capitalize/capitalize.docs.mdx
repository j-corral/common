import { Capitalize } from './capitalize';

### Component usage

```js
import { Capitalize } from '@hoomies/fuel.functions.capitalize';
<Capitalize>hello world!</Capitalize>;
```

### Using props to customize the text

Modify the text to see it change live:

```js live
<Capitalize full={true}>hello world!</Capitalize>
```
